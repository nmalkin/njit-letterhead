% Redefine the \opening environment to position the logo alongside the address
% Also moves the date to the left for consistency with the school's prevailing style
% Inspired by https://tex.stackexchange.com/a/172786
% Original macro source from https://mirrors.ctan.org/macros/latex/base/letter.dtx
\makeatletter
\renewcommand*\opening[1]{%

\vspace*{-0.5in} % Smaller margin on first page; seems to work with values >=1in, but not this one
\ifx\@empty\fromaddress
  \thispagestyle{firstpage}%
    \@date
  \else
    \thispagestyle{empty}%
    \parbox[t]{.7\linewidth}
    {
        \begin{flushleft}
            \includegraphics[width=1.85in]{logo.png}\\
        \end{flushleft}
    }
    \parbox[t]{.3\linewidth}
    {
        \begin{flushleft}%
            {\fontsize{9}{11}
            \ebgaramond
                \fromaddress \\*[2\parskip]%
            }
        \end{flushleft}
    }
  \fi

  \vspace{1\parskip}

  \@date 

  {\raggedright \toname \expandafter\\\toaddress \par}


  #1\par\nobreak}
\makeatother
% End of \opening redefinition
